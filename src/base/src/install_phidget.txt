Phidget lib installieren

sudo apt update
sudo apt install build-essential libusb-1.0-0-dev wget
cd ~/Downloads
wget https://www.phidgets.com/downloads/phidget21/libraries/linux/libphidget/libphidget_2.1.9.20190409.tar.gz
tar -xvf libphidget_2.1.9.20190409.tar.gz
cd libphidget_2.1.9.20190409

./configure
make
sudo make install

sudo ldconfig
